### 方法 2: 使用批处理脚本

创建一个批处理文件 `copy_to_thesis.bat`，内容如下：

```batch
@echo off
echo 正在复制论文项目文件...

set SOURCE=D:\Trae CN\实验报告\1
set DEST=D:\Trae CN\论文

REM 创建目录结构
mkdir "%DEST%\src\fortran"
mkdir "%DEST%\src\python"
mkdir "%DEST%\backups\timestamped"
mkdir "%DEST%\backups\aliyun"
mkdir "%DEST%\docs\papers"
mkdir "%DEST%\docs\experiments"
mkdir "%DEST%\scripts\backup"

REM 复制根目录文件
copy "%SOURCE%\README.md" "%DEST%\"
copy "%SOURCE%\项目使用快速指南.md" "%DEST%\"
copy "%SOURCE%\fortran_env.bat" "%DEST%\"
copy "%SOURCE%\python_env.bat" "%DEST%\"

REM 复制源代码
copy "%SOURCE%\src\fortran\*.*" "%DEST%\src\fortran\"
copy "%SOURCE%\src\python\*.*" "%DEST%\src\python\"

REM 复制备份脚本
copy "%SOURCE%\scripts\backup\*.*" "%DEST%\scripts\backup\"

REM 复制文档
copy "%SOURCE%\docs\*.md" "%DEST%\docs\"
copy "%SOURCE%\docs\experiments\*.*" "%DEST%\docs\experiments\"

echo 复制完成！
pause
```
